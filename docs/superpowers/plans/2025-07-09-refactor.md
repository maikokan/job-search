- [ ] **Step 2: Write migration script**

Create `scripts/migrate_to_merged_db.py` that:
1. Opens `data/jobs.db` and `data/rejections.db`
2. Reads all rows from `job_rejections` table in rejections.db
3. Inserts them into `jobs` table with `status='rejected'`
4. Renames old DBs with `.bak` suffix (or deletes if user confirms)
5. Creates new merged DB with WAL mode

```python
"""Migrate from two-DB setup to single merged DB with WAL mode."""
import sqlite3
from pathlib import Path

def migrate():
    jobs_db = Path("data/jobs.db")
    rej_db = Path("data/rejections.db")
    merged_db = Path("data/jobs_merged.db")

    if not jobs_db.exists():
        print("data/jobs.db not found, nothing to migrate")
        return

    # Create merged DB
    conn = sqlite3.connect(str(merged_db))
    conn.execute("PRAGMA journal_mode=WAL")

    conn.execute('''
        CREATE TABLE IF NOT EXISTS jobs (
            id TEXT PRIMARY KEY,
            title TEXT, company TEXT, location TEXT, url TEXT,
            job_type TEXT, is_remote INTEGER, description TEXT,
            summary TEXT, industry TEXT, gics_code INTEGER,
            gics_confidence REAL, fit_score REAL, enriched_at TEXT,
            notified INTEGER DEFAULT 0, scraped_at TEXT,
            status TEXT DEFAULT 'active',
            rejection_reason TEXT, rejection_detail TEXT
        )
    ''')
    conn.execute("CREATE INDEX IF NOT EXISTS idx_jobs_status ON jobs(status)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_jobs_scraped_at ON jobs(scraped_at DESC)")
    conn.commit()

    # Migrate active jobs
    if jobs_db.exists():
        src = sqlite3.connect(str(jobs_db))
        src.execute("ATTACH DATABASE ? AS rej", (str(rej_db),))
        src.execute("PRAGMA journal_mode=WAL")
        for row in src.execute("SELECT * FROM jobs"):
            conn.execute("INSERT OR IGNORE INTO jobs VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)", row)
        src.close()

    # Migrate rejections
    if rej_db.exists():
        src = sqlite3.connect(str(rej_db))
        for row in src.execute("SELECT * FROM job_rejections"):
            # Append status + rejection fields
            conn.execute("INSERT OR IGNORE INTO jobs VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)", row + ("rejected", None, None))
        src.close()

    conn.close()
    print("Migration complete: data/jobs_merged.db created")
```

def setup_database(db_path: str) -> sqlite3.Connection:
    db_path = Path(db_path).expanduser()
    db_path.parent.mkdir(parents=True, exist_ok=True)

    conn = sqlite3.connect(str(db_path))
    conn.execute("PRAGMA journal_mode=WAL")

    conn.execute('''
        CREATE TABLE IF NOT EXISTS jobs (
            id TEXT PRIMARY KEY, title TEXT, company TEXT, location TEXT, url TEXT,
            job_type TEXT, is_remote INTEGER, description TEXT,
            summary TEXT, industry TEXT, gics_code INTEGER,
            gics_confidence REAL, fit_score REAL, enriched_at TEXT,
            notified INTEGER DEFAULT 0, scraped_at TEXT,
            status TEXT DEFAULT 'active',
            rejection_reason TEXT, rejection_detail TEXT
        )
    ''')
    conn.execute("CREATE INDEX IF NOT EXISTS idx_jobs_status ON jobs(status)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_jobs_scraped_at ON jobs(scraped_at DESC)")

    # Migration: add columns if upgrading
    for col, coldef in [
        ('status', "TEXT DEFAULT 'active'"),
        ('rejection_reason', 'TEXT'),
        ('rejection_detail', 'TEXT'),
    ]:
        try:
            conn.execute(f"ALTER TABLE jobs ADD COLUMN {col} {coldef}")
        except sqlite3.OperationalError:
            pass

    conn.commit()
    return conn
```

from dataclasses import dataclass

@dataclass
class Config:
    desired_gics: list[int]
    rejected_gics: list[int]
    reject_words: list[str]
    ai_endpoint: str
    ai_api_key: str
    no_ai: bool

def validate_config(config: dict, no_ai: bool) -> Config:
    desired_gics = [int(g) for g in config.get("desired_gics", []) if g]
    rejected_gics = [int(g) for g in config.get("rejected_gics", []) if g]
    reject_words = config.get("reject_words", [])
    ai_endpoint = config.get("ai", {}).get("endpoint", "")
    ai_api_key = config.get("ai", {}).get("api_key", "")

    if desired_gics:
        if no_ai:
            # skip GICS filter entirely
            pass
        elif not ai_endpoint:
            raise ValueError("desired_gics is set but ai.endpoint is empty — cannot classify GICS")

    return Config(
        desired_gics=desired_gics,
        rejected_gics=rejected_gics,
        reject_words=reject_words,
        ai_endpoint=ai_endpoint,
        ai_api_key=ai_api_key,
        no_ai=no_ai,
    )
```

- [ ] **Step 1: Create `src/job_search/pipeline.py`**

Define the Pipeline class that orchestrates all stages. Each method is independently testable:

```python
import logging
from pathlib import Path
from typing import List, Dict

from job_search.scrape import search_all, filter_by_reject_words, filter_location
from job_search.gics import classify_gics_batch, classify_gics_batch_retry
from job_search.enrich import enrich_batch
from job_search.telegram import notify_telegram, format_job_message
from job_search.db import setup_database, store_job, reject_and_remove, prune_old, job_id

logger = logging.getLogger(__name__)


class Pipeline:
    """Job search pipeline. Each method is independently testable."""

    def __init__(self, config: dict, no_ai: bool = False, dry_run: bool = False):
        self.config = config
        self.no_ai = no_ai
        self.dry_run = dry_run
        self._setup_logging()

    def _setup_logging(self):
        logging.basicConfig(
            level=logging.INFO,
            format="%(asctime)s %(levelname)s %(name)s: %(message)s",
        )

    def run(self) -> None:
        """Run full pipeline: scrape → filter → classify → enrich → notify → prune."""
        logger.info("Starting job search pipeline")
        jobs = self.scrape()
        if not jobs:
            logger.info("No jobs found")
            return
        jobs = self.filter(jobs)
        if self.should_classify():
            jobs = self.classify(jobs)
        if self.should_enrich():
            self.enrich(jobs)
        self.notify(jobs)
        self.prune()
        logger.info("Pipeline complete")

    def scrape(self) -> List[Dict]:
        terms = self.config.get("search", {}).get("terms", ["intern"])
        locations = self.config.get("search", {}).get("locations", ["Hong Kong"])
        logger.info(f"Scraping: {len(terms)} terms × {len(locations)} locations")
        return search_all(terms, locations, self.config)

    def filter(self, jobs: List[Dict]) -> List[Dict]:
        """Apply location + title reject word filters."""
        reject_words = self.config.get("reject_words", [])
        passed, _ = filter_location(jobs)
        if reject_words:
            passed, _ = filter_by_reject_words(passed, reject_words)
        logger.info(f"Filtered: {len(passed)}/{len(jobs)} jobs passed filters")
        return passed

    def classify(self, jobs: List[Dict]) -> List[Dict]:
        """Classify jobs by GICS. Modifies jobs in-place. Returns all jobs (unmatched go to enrich)."""
        if self.no_ai:
            return jobs
        ai_config = self.config.get("ai", {})
        desired_gics = set(self.config.get("desired_gics", []))
        rejected_gics = set(self.config.get("rejected_gics", []))

        logger.info(f"Classifying {len(jobs)} jobs via GICS")
        classify_gics_batch(jobs, ai_config)

        # Retry NULL
        null_jobs = [j for j in jobs if not j.get("gics_code")]
        if null_jobs:
            classify_gics_batch_retry(null_jobs, ai_config)

        # Split matched vs rejected
        matched = []
        for j in jobs:
            gics = j.get("gics_code")
            if desired_gics and gics not in desired_gics:
                j["_rejected"] = True
                j["_rejection_reason"] = "gics_no_match"
            elif rejected_gics and gics in rejected_gics:
                j["_rejected"] = True
                j["_rejection_reason"] = "gics_rejected"
            else:
                matched.append(j)

        logger.info(f"GICS classified: {len(matched)}/{len(jobs)} matched")
        return matched

    def should_classify(self) -> bool:
        desired_gics = self.config.get("desired_gics", [])
        return bool(desired_gics) and not self.no_ai

    def should_enrich(self) -> bool:
        return self.should_classify() and not self.no_ai

    def enrich(self, jobs: List[Dict]) -> None:
        logger.info(f"Enriching {len(jobs)} jobs")
        enrich_batch(jobs, self.config, self.config.get("ai", {}))

    def notify(self, jobs: List[Dict]) -> None:
        if not jobs or self.dry_run:
            return
        logger.info(f"Notifying Telegram: {len(jobs)} jobs")
        notify_telegram(jobs)

    def prune(self) -> None:
        if self.dry_run:
            return
        retention = self.config.get("database", {}).get("retention_days", 365)
        logger.info(f"Pruning jobs older than {retention} days")
```
